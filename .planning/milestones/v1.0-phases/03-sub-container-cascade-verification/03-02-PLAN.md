```python
"""Verify SAM Lambda runtime sub-containers are cleaned up on normal teardown.

Per D-07: runs in a subprocess session that invokes Lambda then exits cleanly.
The parent process polls Docker after subprocess exit to assert no sam_ prefixed
containers remain — confirming _teardown_network properly stops and removes
SAM-spawned sub-containers during normal session teardown.

Only gates on ryuk_disabled (runs on macOS too — this is the normal teardown
path, not the crash path, so Docker Desktop's TCP proxy is not relevant).
"""

from __future__ import annotations

import subprocess
import time
from pathlib import Path

import docker
import pytest
from testcontainers.core.config import testcontainers_config

# Skip only when Ryuk is disabled — normal teardown does not depend on Ryuk
# for cleanup, but the docker_network fixture uses Ryuk registration which
# would fail when disabled. Per D-05/D-09: crash path skips macOS, teardown
# path only skips when ryuk_disabled.
pytestmark = pytest.mark.skipif(
    testcontainers_config.ryuk_disabled,
    reason="docker_network fixture requires Ryuk-enabled environment for registration",
)


def _write_teardown_session(session_dir: Path, *, fixture_dir: Path) -> None:
    """Write a pytest session that starts SAM, invokes Lambda, then exits.

    The session runs a full SAM stack (build + start-api) and triggers a
    Lambda invocation via HTTP GET /hello to create a Lambda runtime
    sub-container. The test then exits normally — session teardown should
    clean up the sub-container via _teardown_network.
    """
    conftest = session_dir / "conftest.py"
    conftest.write_text(f"""\
from __future__ import annotations
from pathlib import Path
import pytest
from samstack.settings import SamStackSettings

# Absolute path to the hello_world test fixture, embedded at generation time.
FIXTURE_DIR = Path(r"{fixture_dir}")

@pytest.fixture(scope="session")
def samstack_settings() -> SamStackSettings:
    return SamStackSettings(
        sam_image="public.ecr.aws/sam/build-python3.13",
        project_root=FIXTURE_DIR,
        log_dir="logs/sam",
        add_gitignore=False,
    )
""")

    test_file = session_dir / "test_invoke_and_exit.py"
    test_file.write_text("""\
from __future__ import annotations
import time
import requests


def test_invoke_lambda_then_exit(sam_api: str) -> None:
    # Trigger Lambda invocation via HTTP GET /hello.
    # This creates a Lambda runtime sub-container on the Docker network.
    resp = requests.get(f"{sam_api}/hello", timeout=30)
    assert resp.status_code == 200
    # Brief sleep to ensure Docker daemon has processed the container creation.
    time.sleep(3)
    # Test exits normally — session teardown runs _teardown_network which
    # should stop and remove the Lambda sub-container.
""")


def _poll_for_containers(
    client: docker.DockerClient,
    name_prefix: str,
    expect_gone: bool,
    timeout: float = 15.0,
    interval: float = 0.5,
) -> None:
    """Poll Docker until containers are gone (expect_gone=True) or found."""
    deadline = time.monotonic() + timeout
    while time.monotonic() < deadline:
        remaining = client.containers.list(
            all=True, filters={"name": name_prefix}
        )
        if expect_gone and not remaining:
            return  # Success: containers are gone
        if not expect_gone and remaining:
            return  # Success: containers found
        time.sleep(interval)
    # Timeout — raise assertion with diagnostic information
    state = client.containers.list(all=True, filters={"name": name_prefix})
    names = [c.name for c in state]
    if expect_gone:
        raise AssertionError(
            f"Expected zero containers matching '{name_prefix}' after "
            f"teardown, but {len(state)} remain after {timeout}s: {names}"
        )
    raise AssertionError(
        f"Expected containers matching '{name_prefix}' but none found "
        f"after {timeout}s"
    )
```

**Implementation details:**
- `_write_teardown_session` mirrors Plan 01's `_write_subprocess_session` but generates a test that exits normally (no stall) instead of stalling for SIGKILL
- `_poll_for_containers` is a general-purpose polling helper with `expect_gone` flag — the test uses it to assert containers are gone, but the helper could also be used to verify containers exist in other tests
- The conftest sets `project_root=FIXTURE_DIR` to the hello_world fixture and does NOT set `sam_env_vars` — the Lambda handler's GET path doesn't need `TEST_BUCKET`
- Per D-05: no `sys.platform == "darwin"` skip — normal teardown runs everywhere
- Per D-09: gate behind `testcontainers_config.ryuk_disabled` — the docker_network fixture requires Ryuk for registration even though normal teardown doesn't depend on Ryuk for cleanup
- Per D-04: default timeout is 15s for normal teardown path
- Per D-03: use `containers.list(all=True, ...)` to catch containers in any state
- Per RESEARCH.md finding #5: use `name` filter (`sam_`), not `network` filter

```python
class TestSubcontainerNormalTeardown:
    def test_subcontainers_cleaned_after_normal_teardown(
        self, tmp_path: Path
    ) -> None:
        """Sub-containers are removed after normal pytest session teardown.

        Launches a subprocess pytest session that starts SAM API, invokes
        Lambda via HTTP GET /hello (creating a Lambda runtime sub-container),
        and exits normally. After the subprocess completes, polls Docker to
        assert zero containers with name prefix "sam_" remain — confirming
        _teardown_network properly cleaned up SAM-spawned sub-containers
        during normal session teardown.
        """
        session_dir = tmp_path / "teardown_session"
        session_dir.mkdir()

        fixture_dir = Path.cwd() / "tests" / "fixtures" / "hello_world"
        _write_teardown_session(session_dir, fixture_dir=fixture_dir)

        # Launch subprocess pytest session.
        # Use DEVNULL to avoid pipe-buffer deadlocks (same pattern as TEST-03).
        # Timeout of 180s allows for SAM build + LocalStack pull + Lambda invoke
        # on slow CI runners.
        proc = subprocess.Popen(
            ["uv", "run", "pytest", str(session_dir), "-v", "--timeout=180"],
            stdout=subprocess.DEVNULL,
            stderr=subprocess.DEVNULL,
        )

        # Wait for normal completion — exit code 0 means the session ran
        # successfully and all fixture teardowns completed without errors.
        exit_code = proc.wait(timeout=300)

        assert exit_code == 0, (
            f"Subprocess pytest session failed with exit code {exit_code}. "
            "The SAM session did not complete normally — check that LocalStack "
            "is accessible and the hello_world fixture builds correctly."
        )

        # Poll Docker for sam_ prefixed containers.
        # After normal teardown, _teardown_network should have stopped and
        # removed all Lambda sub-containers. Per D-04: 15s timeout for normal
        # teardown (shorter than crash path because Docker operations are
        # synchronous during normal exit, not async Ryuk detection).
        docker_client = docker.from_env()
        _poll_for_containers(
            docker_client,
            name_prefix="sam_",
            expect_gone=True,
            timeout=15.0,
            interval=0.5,
        )
```

**Implementation details:**
- The test class `TestSubcontainerNormalTeardown` follows the same naming convention as `TestRyukCrashCleanup` in `test_ryuk_crash.py`
- `fixture_dir` is computed as `Path.cwd() / "tests" / "fixtures" / "hello_world"` — the subprocess runs from the project root, and the generated conftest embeds this as an absolute path
- Subprocess uses `--timeout=180` — 3 minutes is enough for SAM build + LocalStack pull + Lambda invocation even on slow CI
- `proc.wait(timeout=300)` — 5 minutes total timeout for the subprocess (worst case: all images need pulling)
- Per D-04: teardown assertion uses 15s timeout (vs 30s for crash path)
- `_poll_for_containers` with `expect_gone=True` handles the assertion internally — raises AssertionError with remaining container names if timeout expires
- Per D-05: no macOS skip — `pytestmark` only checks `ryuk_disabled`, so this test runs on macOS through Docker Desktop

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Docker daemon socket | Test code calls `docker.from_env()` which connects to the Docker socket at `/var/run/docker.sock`. Container listing queries cross this boundary. |
| Subprocess boundary | `subprocess.Popen` launches a separate pytest session. The parent reads exit code and polls Docker independently — no stdout/stderr parsing. |
| Filesystem (subprocess conftest) | Generated conftest and test files are written to `tmp_path` using absolute paths for fixture directories. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-05 | Tampering | Generated conftest/test files in `tmp_path` | accept | Files are generated by the test itself with no external input. The `fixture_dir` is computed as `Path.cwd() / "tests" / "fixtures" / "hello_world"` — a project-relative path, not user-controlled. Accept. |
| T-03-06 | Information Disclosure | Docker container metadata (names, status) queried via `containers.list()` | accept | Container names and statuses are not sensitive. No credentials, environment variables, or secrets are exposed. Accept. |
| T-03-07 | Denial of Service | Subprocess launch + Docker image pull | accept | Subprocess timeout is bounded (300s for `proc.wait`, 180s for pytest `--timeout`). Docker image pull is a one-time cost (cached after first run). Accept. |
| T-03-08 | Elevation of Privilege | Subprocess Docker socket access inherited from parent | accept | The child pytest session accesses the same Docker socket as the parent — this is by design (Docker-in-Docker for SAM). No privilege boundary is crossed. Accept. |
</threat_model>
