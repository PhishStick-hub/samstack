    ```python
    """Integration crash test: confirms Ryuk removes the Docker network after SIGKILL.

    TEST-03 per requirements. Requires Docker + Ryuk enabled (not skipped in CI
    where TESTCONTAINERS_RYUK_DISABLED=true).
    """

    from __future__ import annotations

    import os
    import signal
    import subprocess
    import time
    from pathlib import Path

    import docker
    import docker.errors
    import pytest

    from testcontainers.core.config import testcontainers_config


    # Skip the entire module when Ryuk is disabled (CI environments).
    pytestmark = pytest.mark.skipif(
        testcontainers_config.ryuk_disabled,
        reason="Ryuk disabled — crash-cleanup test requires Ryuk",
    )


    def _write_subprocess_session(session_dir: Path) -> None:
        """Write a minimal pytest session that creates docker_network and stalls."""
        conftest = session_dir / "conftest.py"
        conftest.write_text(
            """\
    from __future__ import annotations

    import pytest
    from samstack.settings import SamStackSettings


    @pytest.fixture(scope="session")
    def samstack_settings() -> SamStackSettings:
        # Minimal settings — SAM image is required by SamStackSettings but
        # docker_network does not use it. No SAM build or LocalStack pull occurs.
        return SamStackSettings(sam_image="public.ecr.aws/sam/build-python3.13")
    """
        )

        test_file = session_dir / "test_stall.py"
        test_file.write_text(
            """\
    from __future__ import annotations

    import time


    def test_stall(docker_network: str) -> None:
        # Network is created; stall so SIGKILL arrives while the fixture is live.
        time.sleep(60)
    """
        )


    def _poll_until_gone(
        client: docker.DockerClient,
        network_name: str,
        timeout: float = 5.0,
        interval: float = 0.5,
    ) -> bool:
        """Return True if the network disappears within timeout seconds."""
        deadline = time.monotonic() + timeout
        while time.monotonic() < deadline:
            try:
                client.networks.get(network_name)
            except docker.errors.NotFound:
                return True
            time.sleep(interval)
        return False


    class TestRyukCrashCleanup:
        def test_network_removed_after_sigkill(self, tmp_path: Path) -> None:
            """Ryuk removes the samstack bridge network when pytest is SIGKILLed."""
            session_dir = tmp_path / "crash_session"
            session_dir.mkdir()
            _write_subprocess_session(session_dir)

            # Launch subprocess pytest session.
            proc = subprocess.Popen(
                ["uv", "run", "pytest", str(session_dir), "-v", "--timeout=120"],
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
            )

            # Give the subprocess time to create the network and register with Ryuk.
            # docker_network runs at session start; 3 s is ample.
            time.sleep(3)

            # Read the network name from Docker: find the samstack-* network
            # whose labels include the testcontainers session-id.
            docker_client = docker.from_env()
            samstack_networks = docker_client.networks.list(
                filters={"label": "org.testcontainers.session-id"}
            )
            assert samstack_networks, (
                "No labeled samstack network found — did docker_network run? "
                "Increase the sleep above or check subprocess output."
            )
            # There should be exactly one samstack network from this subprocess.
            network_name = samstack_networks[0].name

            # SIGKILL the subprocess — simulates a crashed pytest process.
            os.kill(proc.pid, signal.SIGKILL)
            proc.wait()

            # Poll Docker until Ryuk removes the network (hard assert per D-10).
            gone = _poll_until_gone(docker_client, network_name, timeout=5.0, interval=0.5)

            # Document sub-container cascade observation (not hard-asserted per D-10).
            try:
                leftover_containers = docker_client.containers.list(
                    filters={"network": network_name}
                )
                cascade_note = (
                    f"Sub-containers still attached after network removal: "
                    f"{[c.name for c in leftover_containers]}"
                    if leftover_containers
                    else "No sub-containers remained (cascade occurred or none created)"
                )
            except Exception:
                cascade_note = "Could not enumerate sub-containers (network already gone)"

            # Print for test output visibility.
            print(f"\n[TEST-03] Sub-container cascade: {cascade_note}")

            assert gone, (
                f"Docker network '{network_name}' still exists {5.0}s after SIGKILL. "
                "Ryuk did not clean it up. Verify TESTCONTAINERS_RYUK_DISABLED is not set."
            )
    ```
